library(RMySQL) con = dbConnect(MySQL(),user='root', password='admin', dbname='pratik', host='localhost') dbSendQuery(con, "USE pratik") dbSendQuery(con, "CREATE TABLE authors (author_id INT AUTO_INCREMENT PRIMARY KEY, author_last VARCHAR(50), author_first VARCHAR(50), country VARCHAR(50));") # Adding data into tables dbSendQuery(con, "INSERT INTO authors (author_last, author_first, country) VALUES('Kumar','Manoj','India');") dbSendQuery(con, "INSERT INTO authors (author_last, author_first, country) VALUES('deo','pratik','germany');") dbSendQuery(con, "INSERT INTO authors (author_last, author_first, country) VALUES('pakerappa','suresh','england');") # Updating the data dbSendQuery(con, "update authors set country = 'singapour' where author_first = 'suresh'") # Deleting the row dbSendQuery(con, "DELETE from authors WHERE country = 'singapour'")
